import gradio as gr
import numpy as np
from sklearn.datasets import load_iris
from sklearn.ensemble import RandomForestClassifier

# ‚îÄ‚îÄ Train model ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
iris = load_iris()
clf  = RandomForestClassifier(n_estimators=100, random_state=42)
clf.fit(iris.data, iris.target)
labels = iris.target_names

# ‚îÄ‚îÄ Prediction function ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
def predict_flower(sepal_len, sepal_wid, petal_len, petal_wid):
    X = np.array([[sepal_len, sepal_wid, petal_len, petal_wid]])
    pred  = clf.predict(X)[0]
    proba = clf.predict_proba(X)[0]
    return {labels[i]: float(proba[i]) for i in range(len(labels))}

# ‚îÄ‚îÄ Build UI ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
app = gr.Interface(
    fn      = predict_flower,
    inputs  = [
        gr.Slider(4.0, 8.0, value=5.8, label="Sepal Length (cm)"),
        gr.Slider(2.0, 4.5, value=3.0, label="Sepal Width (cm)"),
        gr.Slider(1.0, 7.0, value=4.0, label="Petal Length (cm)"),
        gr.Slider(0.1, 2.5, value=1.2, label="Petal Width (cm)"),
    ],
    outputs = gr.Label(num_top_classes=3),
    title   = "üå∏ Iris Flower Classifier",
    description = "Adjust sliders to classify iris species with Random Forest.",
    examples = [[5.1,3.5,1.4,0.2], [6.7,3.1,4.7,1.5], [6.3,3.3,6.0,2.5]]
)

app.launch()
